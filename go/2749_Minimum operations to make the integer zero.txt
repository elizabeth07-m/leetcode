func makeTheIntegerZero(num1 int, num2 int) int {
    if num1 < num2 {
        return -1
    }

    var res = 1
    for {
        num1 -= num2
        if num1 <= 0 {
            return -1
        }
        c := bitCount(num1)
        if num1 >= res && c <= res {
            return res
        }
        res++
    }
    return res
}

func bitCount(n int) int {
    var c int
    for n > 0 {
        n = n & (n-1)
        c++
    }
    return c
}
